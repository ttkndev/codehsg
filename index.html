Kho đề thi Học liệu 
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<int> a(n);
    for (int i = 0; i < n; i++)
        cin >> a[i];

    sort(a.begin(), a.end());

    cout << "Sorted: ";
    for (int x : a)
        cout << x << " ";

    return 0;
}
